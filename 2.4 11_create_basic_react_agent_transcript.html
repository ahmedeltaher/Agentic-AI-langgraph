📋 الكود الكامل

تعالوا نشوف الكود كله مجمع:

# Imports from langchain_openai import AzureChatOpenAI from langgraph.prebuilt import create_react_agent from langchain_core.messages import AIMessage, HumanMessage, SystemMessage from langchain_core.tools import tool import os # Setup Azure OpenAI os.environ["AZURE_OPENAI_API_KEY"] = "your-api-key" os.environ["AZURE_OPENAI_ENDPOINT"] = "your-endpoint" # Create LLM llm = AzureChatOpenAI( azure_deployment="gpt-4o", api_version="2024-05-01-preview", temperature=0 ) # Tools (created earlier) @tool def find_sum(x: int, y: int) -> int: """Add two numbers and return their sum.""" return x + y @tool def find_product(x: int, y: int) -> int: """Multiply two numbers and return their product.""" return x * y # Setup tools list agent_tools = [find_sum, find_product] # Define system prompt system_prompt = SystemMessage(content=""" You are a helpful math assistant. You can only solve addition and multiplication problems. You must ALWAYS use the provided tools for calculations. Never perform calculations directly. """) # Create the ReAct agent agent_graph = create_react_agent( model=llm, tools=agent_tools, state_modifier=system_prompt )
